def daemonize
        fork do
                Process.setsid
                exit if fork
                Dir.chdir('/tmp')
                STDIN.reopen('/dev/null')
                STDOUT.reopen('/dev/null', 'a')
                STDERR.reopen('/dev/null', 'a')

                trap("TERM") { 
                  logger.debug("Caught TERM signal") 
                  exit
                }
                yield
        end
end
