  def initialize(port, host, xml, logger, timeout=10, debug=false)
    @logger = logger
    @connections = []
    @@connMutex = Mutex.new
    @@clientsMutex = Mutex.new
    @@bogusclients = 0
    @@totalclients = 0
    @timeout = timeout
    @@starttime = Time.new
    @xml = xml
    @port = port
    @host = host

    if debug
      @logger.level = Logger::DEBUG
      debug("Set logging level to #{@logger.level}")
    else
      @logger.level = Logger::INFO
    end
  end
